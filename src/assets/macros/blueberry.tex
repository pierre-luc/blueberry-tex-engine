

{% macro cellColor(cell) %}
	{% if col.bgColor %}
		{{ "\cellcolor{" + col.bgColor + "}" | raw }}
	{% endif %}
	{% if col.color %}
		{ {{ "\textcolor{" + col.color + "}" | raw }}{{cell.value | raw}} }
	{% else %}
		{{cell.value | raw}}
	{% endif %}
{% endmacro %}

{% macro cellValue( cell ) %}
	{% if cell.align %}
		{% set align = {left: 'l', center: 'c', right: 'r' }%}
		\multicolumn{1}{|{{align[cell.align]}}|}{ {{cellColor(cell)}} }&
	{% else %}
		{{cellColor(cell)}}&
	{% endif %}
{% endmacro %}

{% macro table( p )%}
	{% set maxRow = 0 %}
	{% for row in p -%}
		{%- if row.type == 'header' -%}
			{% if row.cells.length > maxRow.length %}
				{% set maxRow = row.cells %}
			{% endif %}
		{% else -%}
			{% if row.cells.length > maxRow.length %}
				{% set maxRow = row %}
			{% endif %}
		{% endif %}
	{%- endfor %}

	\begin{tabular}[t]{ {% for row in maxRow -%}|c{%- endfor %}l|}
		\hline
		{% for row in p %}
			{%- if row.type == 'header' -%}
				\rowcolor{secondarycolor}
				{% for col in row.cells -%}
					{\textcolor{white} {{ col | raw  }}}&
				{%- endfor %}\\
				\hline
			{% else -%}
				{% for col in row -%}
					{% if col.value %}
						{{cellValue(col)}}
					{% else %}
						{{ col | raw  }}&
					{% endif %}
				{%- endfor %}\\
			{% endif %}
		{% endfor %}
		\hline
	\end{tabular}
{% endmacro %}
